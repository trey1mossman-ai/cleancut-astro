Tech JavaScript Behavior

<script>
  // Get elements
  const modal = document.getElementById('estimate-modal');
  const backdrop = document.getElementById('estimate-backdrop');
  const formContainer = document.getElementById('estimate-form-container');
  const closeBtn = document.getElementById('close-estimate-modal');
  const form = document.getElementById('estimate-form');
  const successState = document.getElementById('estimate-success');
  
  // Open triggers - ALL elements with data-open-estimate attribute
  const openTriggers = document.querySelectorAll('[data-open-estimate]');
  
  function openModal() {
    modal.classList.remove('hidden');
    // Animate in
    setTimeout(() => {
      formContainer.classList.remove('scale-95', 'opacity-0');
      formContainer.classList.add('scale-100', 'opacity-100');
    }, 10);
    document.body.style.overflow = 'hidden';
  }
  
  function closeModal() {
    formContainer.classList.remove('scale-100', 'opacity-100');
    formContainer.classList.add('scale-95', 'opacity-0');
    setTimeout(() => {
      modal.classList.add('hidden');
      document.body.style.overflow = '';
      // Reset form after close
      form.reset();
      form.classList.remove('hidden');
      successState.classList.add('hidden');
    }, 300);
  }
  
  // Attach open triggers
  openTriggers.forEach(trigger => {
    trigger.addEventListener('click', (e) => {
      e.preventDefault();
      openModal();
    });
  });
  
  // Close handlers
  closeBtn.addEventListener('click', closeModal);
  backdrop.addEventListener('click', closeModal);
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && !modal.classList.contains('hidden')) {
      closeModal();
    }
  });
  
  // Form submit
  form.addEventListener('submit', async (e) => {
    e.preventDefault();
    // TODO: Add actual form submission logic
    // For now, show success state
    form.classList.add('hidden');
    successState.classList.remove('hidden');
    
    // Auto-close after 3 seconds
    setTimeout(closeModal, 3000);
  });
</script>
